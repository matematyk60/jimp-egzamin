\section{Base class with functions set as virtual}
\begin{multicols}{2}
\Large
\begin{minted}{c}
class Base{
public:
    Base(){
        cout << "Base()\n";
    }
    virtual void Print() {
        cout << "Base\n";
    }
    virtual ~Base(){
        cout << "~Base\n";
    }
};

class Derrived : public Base{
public:
    Derrived(){
        cout << "Derrived()\n";
    }
    void Print(){
        cout << "Derrived\n";
    }

    ~Derrived(){
        cout << "~Derrived\n";
    }
};


\end{minted}
\subsection{Creating reference to Derrived}
\begin{minted}{c}
int main(){
    Derrived d;
    d.Print();
}
out:
Base()
Derrived()
Derrived
~Derrived
~Base
\end{minted}
\subsection{Creating pointer to Derrived}
\begin{minted}{c}
int main(){
    Derrived* d = new Derrived;
    d->Print();
    delete d;
}
out:
Base()
Derrived()
Derrived
~Derrived
~Base
\end{minted}
\subsection{Creating pointer to Base}
\begin{minted}{c}
int main(){
    Base* d = new Derrived;
    d->Print();
    delete d;
}
out:
Base()
Derrived()
Derrived
~Derrived
~Base
\end{minted}
\end{multicols}
